window.addSlot = async function() { const prof = document.getElementById('professorSelect').value; const dia = document.getElementById('slotDia').value; const hora = document.getElementById('slotHora').value; if(!prof){ alert("Selecione um professor"); return; } // Criamos um ID que inclua o professor para evitar sobreposição const id = `${prof}_${dia}_${hora.replace(':','')}`; await setDoc(doc(db, "disponibilidade", id), { teacherId: prof, dia: dia, hora: hora, status: "livre" }); loadSlots(); } window.addMultipleSlots = async function() { const prof = document.getElementById('professorSelect').value; const dia = document.getElementById('slotDia').value; const inicio = document.getElementById('slotInicio').value; const fim = document.getElementById('slotFim').value; if(!prof){ alert("Selecione o professor"); return; } let curr = new Date(`2025-01-01T${inicio}`); const end = new Date(`2025-01-01T${fim}`); while(curr < end){ const hh = String(curr.getHours()).padStart(2,'0'); const mm = String(curr.getMinutes()).padStart(2,'0'); const timeStr = `${hh}:${mm}`; const id = `${prof}_${dia}_${hh}${mm}`; await setDoc(doc(db, "disponibilidade", id), { teacherId: prof, dia: dia, hora: timeStr, status: "livre" }); curr.setMinutes(curr.getMinutes() + 30); } loadSlots(); }
